16  public class Kayttoliittyma {
17  
18      private Pelilauta lauta;
19      private Boolean voitto;
20      private Boolean pelinLoppuminen;
21      private Scanner scan;
22  
23      public Kayttoliittyma() {
24          this.voitto = false;
25          this.pelinLoppuminen = false;
26          this.scan = new Scanner(System.in);
27          pelaa();
28      }
29  
30      public void pelaa() {
31          boolean totta = false;
32          int laudanKoko = 0;
33          System.out.println("Anna laudan koko YxY. Oltava vähintään Y=3:");
34          while (!totta) {
35              laudanKoko = scan.nextInt();
36              if (laudanKoko < 3) {
37                  System.out.println("Laudan oltava isompi! Anna uusi koko: ");
38                  totta = false;
39              } else {
40                  totta = true;
41              }
42          }
43          lauta = new Pelilauta(laudanKoko);
44  
45          lauta.tulostaPelilauta();
46          System.out.println(lauta.miinojaYhteensa());
47      }
48  
49      public void pelaaVuoro() {
50          System.out.println("Anna avattavan ruudun rivi sijainti:");
51          int sijaintiRivi = scan.nextInt();
52          System.out.println("Anna avattavan ruudun sarake sijainti:");
53          int sijaintiSarake = scan.nextInt();
54  
55          avaaRuutu(sijaintiRivi, sijaintiSarake);
56      }
57  
58      public void avaaRuutu(int rivi, int sarake) {
59          Ruutu[][] ruudut = lauta.getRuudut();
60      }
