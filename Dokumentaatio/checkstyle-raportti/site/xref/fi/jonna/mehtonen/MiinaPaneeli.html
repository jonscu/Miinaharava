48      /**
49       * Luodaan pelipohja ja nappulat.
50       *
51       */
52      public void luoNappulat() {
53          paneeli.setLayout(new GridLayout(koko, koko));
54          nappulat = new JButton[koko][koko];
55  
56          for (int k = 0; k < koko; k++) {
57              for (int i = 0; i < koko; i++) {
58                  JButton nappula = new JButton();
59                  nappula.addMouseListener(new Mouse());
60                  nappulat[k][i] = nappula;
61                  paneeli.add(nappulat[k][i]);
62              }
63          }
64          add(paneeli);
65      }

106     /**
107      * Avataan ruutu hiiren vasemmalla näppäimellä. Tällä hetkellä '*' on miina,
108      * ja lukuarvo on naapurissa olevien miinojen määrä.
109      *
110      * @param nappula JButton, jota on klikattu.
111      * @param rivi nappulan rivi sijainti.
112      * @param sarake nappulan sarake sijainti.
113      */
114     public void merkataanRuutuAvatuksi(JButton nappula, int rivi, int sarake) {
115         ruudut[rivi][sarake].merkkaaRuutuAvatuksi();
116         if (ruudut[rivi][sarake].isOnkoMiina()) {
117             nappula.setText("*");
118         } else {
119             String luku = Integer.toString(ruudut[rivi][sarake].getArvo());
120             nappula.setIcon(null);
121             nappula.setText(luku);
122         }
123         nappula.setEnabled(false);
124     }

182     public class Logiikka {
183 
184         /**
185          * Tarkistetaan, onko peli voitettu. Peli on voitettu, jos kaikki
186          * miinattomat ruudut ovat avattu.
187          *
188          * @return true, jos peli on voitettu, muulloin false.
189          */
190         public Boolean voitto() {
191             int ruutujaAvattu = 0;
192 
193             for (int k = 0; k < koko; k++) {
194                 for (int j = 0; j < koko; j++) {
195                     if (ruudut[k][j].isOnkoAvattu() && !ruudut[k][j].isOnkoMiina()) {
196                         ruutujaAvattu++;
197                     }
198                 }
199             }
200             if (ruutujaAvattu == (koko * koko) - miinoja) {
201                 return true;
202             } else {
203                 return false;
204             }
205         }
206 
207         /**
208          * Tarkistetaan, onko peli hävitty. Peli havitaan, jos kayttaja avaa
209          * ruudun, jossa on miina.
210          *
211          * @return true, jos peli hävittiin, muulloin false.
212          */
213         public Boolean havio() {
214             for (int k = 0; k < koko; k++) {
215                 for (int j = 0; j < koko; j++) {
216                     if (ruudut[k][j].isOnkoMiina() && ruudut[k][j].isOnkoAvattu()) {
217                         return true;
218                     }
219                 }
220             }
221             return false;
222         }
223 
224         /**
225          * Tarkistetaan painettiinko oikeeta vai vasenta hiiren nappulaa. Jos
226          * klikattiin vasenta nappulaa, merkataan ruutu avatuksi ja
227          * tarkistetaan, voitettiinko tai havittiinko peli. Jos klikattiin
228          * oikeata nappulaa, niin laitetaan ruutuun lippu, mikali siina ei
229          * ennestaan ollut lippua. Mikäli oli, niin lippu otetaan pois, jolloin
230          * nappi on avaamaton.
231          *
232          * @param tapahtuma Hiiren klikkauksesta aiheutuva MouseEvent.
233          */
234         public void avataanRuutu(MouseEvent tapahtuma) {
235             Component nappula = tapahtuma.getComponent();
236             if (nappula.isEnabled() && tapahtuma.getButton() == 1) {
237                 for (int k = 0; k < koko; k++) {
238                     for (int j = 0; j < koko; j++) {
239                         if (nappula == nappulat[k][j]) {
240                             if (ruudut[k][j].isOnkoMiina() && siirrot == 0 && !ruudut[k][j].isOnkoLippu()) {
241                                 lauta.siirraMiina(k, j);
242                                 merkataanRuutuAvatuksi(nappulat[k][j], k, j);
243                                 avaaNaapurit(nappulat[k][j], k, j, nappulat);
244                                 siirrot++;
245                             } else if (!ruudut[k][j].isOnkoLippu()) {
246                                 merkataanRuutuAvatuksi(nappulat[k][j], k, j);
247                                 avaaNaapurit(nappulat[k][j], k, j, nappulat);
248                                 siirrot++;
249                                 if (voitto()) {
250                                     voitaJaSuljePeli();
251                                 } else if (havio()) {
252                                     haviaJaSuljePeli();
253                                 } else {
254                                     break;
255                                 }
256                             }
257                         }
258                     }
259                 }
260             } else if (tapahtuma.getButton() == 3) {
261                 for (int k = 0; k < koko; k++) {
262                     for (int j = 0; j < koko; j++) {
263                         if (nappula == nappulat[k][j]) {
264                             if (!ruudut[k][j].isOnkoLippu() && lippujaKaytetty != miinoja && !ruudut[k][j].isOnkoAvattu()) {
265                                 lippujaKaytetty++;
266                                 asetaLippu(nappulat[k][j], k, j);
267                                 break;
268                             } else if (ruudut[k][j].isOnkoLippu()) {
269                                 lippujaKaytetty--;
270                                 poistaLippu(nappulat[k][j], k, j);
271                                 break;
272                             }
273                         }
274                     }
275                 }
276 
277             }
278         }

280         /**
281          * Mikali klikatun ruudun yhdessakaan naapuriruudussa ei ole miinaa,
282          * peli paljastaa ensin kaikki siihen yhteydessa olevat ruudut, joiden
283          * naapurissa ei myoskaan ole miinaa, sekä lisaksi taman paljastetun
284          * alueen reunalla olevat ruudut, joiden naapurissa on miina.
285          *
286          * @param nappula JButton, jota on klikattu.
287          * @param rivi nappulan rivi sijainti.
288          * @param sarake nappulan sarake sijainti.
289          * @param nappulat laudan kaikki nappulat matriisissa.
290          */
291         public void avaaNaapurit(JButton nappula, int rivi, int sarake, JButton[][] nappulat) {
292             for (int i = -1; i <= 1; i++) {
293                 for (int k = -1; k <= 1; k++) {
294                     int uusiRivi = rivi - i;
295                     int uusiSarake = sarake - k;
296                     if (uusiRivi == rivi && uusiSarake == sarake) {
297                         continue;
298                     } else if (uusiRivi >= 0 && uusiRivi < koko && uusiSarake >= 0 && uusiSarake < koko) {
299                         if (!ruudut[uusiRivi][uusiSarake].isOnkoAvattu() && ruudut[uusiRivi][uusiSarake].getArvo() == 0 && !ruudut[uusiRivi][uusiSarake].isOnkoMiina()) {
300                             merkataanRuutuAvatuksi(nappulat[uusiRivi][uusiSarake], uusiRivi, uusiSarake);
301                             avaaNaapurit(nappulat[uusiRivi][uusiSarake], uusiRivi, uusiSarake, nappulat);
302                         }
303                     }
304                 }
305             }
306         }
