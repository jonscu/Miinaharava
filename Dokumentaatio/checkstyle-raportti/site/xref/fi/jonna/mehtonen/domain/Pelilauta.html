1   /**
2    * Pelilauta luokka alustaa itse pelilaudan, luomalla ruudut, 
3    * asettamalla niihin arvot ja asettamalla miinat.
4    */
5   package fi.jonna.mehtonen.domain;
6   
7   import fi.jonna.mehtonen.domain.Ruutu;
8   import java.util.Random;
9   
10  /**
11   *
12   * @author jonscu
13   */
14  public class Pelilauta {
15  
16      private int koko;
17      private Ruutu[][] ruudut;
18      private Random rnd;
19  
20      public Pelilauta(int laudanKoko) {
21          this.koko = laudanKoko;
22          this.rnd = new Random();
23          ruudut = new Ruutu[koko][koko];
24          alustaPelilauta();
25      }
26  
27      /** Luodaan ruudut, sijoitetaan randomisti miinat ja asetetaan miinattomille
28       * ruuduille arvot.
29       */
30      public void alustaPelilauta() {
31          luoRuudut();
32          asetaMiinat();
33          asetaRuuduilleArvot();
34      }
35  
36      /** Luodaan matriisi, joka sisältää ruutuja.
37       * 
38       */
39      public void luoRuudut() {
40          for (int k = 0; k < koko; k++) {
41              for (int j = 0; j < koko; j++) {
42                  ruudut[k][j] = new Ruutu();
43              }
44          }
45      }
46  
47      /** Arvotaan miinoille paikat pelissa.
48       * 
49       */
50      public void asetaMiinat() {
51          int i;
52          int j;
53          for (int k = 0; k < (koko * koko) / 5; k++) {
54              i = rnd.nextInt(koko);
55              j = rnd.nextInt(koko);
56  
57              if (ruudut[i][j].isOnkoMiina()) {
58                  k--;
59              } else {
60                  ruudut[i][j].asetaMiina();
61              }
62          }
63      }
64  
65      /** Jos ensimmäinen ruutu, joka avataan sisältää miinan, niin 
66      * miinalle arvotaan uusi paikka ja peli jatkuu. Kayttaja ei huomaa siirtoa.
67      * 
68      * @param rivi kertoo, millä rivillä korvattava miina sijaitsee.
69      * @param sarake kertoo, millä sarakkeella korvattava miina sijaitsee.
70      */
71      public void siirraMiina(int rivi, int sarake) {
72          boolean siirretty = false;
73          while (!siirretty) {
74              int uusiRivi = rnd.nextInt(koko);
75              int uusiSarake = rnd.nextInt(koko);
76              if (!ruudut[uusiRivi][uusiSarake].isOnkoMiina()) {
77                  ruudut[rivi][sarake].poistaMiina();
78                  ruudut[uusiRivi][uusiSarake].asetaMiina();
79                  asetaRuuduilleArvot();
80                  siirretty = true;
81              }
82          }
83      }
84  
85      /** Asetetaan ruuduille arvoiksi se luku, joka vastaa naapuriruuduissa olevien
86      * miinojen maara.
87      */
88      public void asetaRuuduilleArvot() {
89          for (int k = 0; k < koko; k++) {
90              for (int j = 0; j < koko; j++) {
91                  if (!ruudut[k][j].isOnkoMiina()) {
92                      ruudut[k][j].setArvo(montaMiinaaNaapurissa(k, j));
93                  }
94              }
95          }
96  
97      }
98  
99      /** Lasketaan, kuinka monta miinaa on ruudun naapureissa.
100      * 
101      * @param rivi kertoo, millä rivillä kyseinen ruutu sijaitsee.
102      * @param sarake kertoo, millä sarakkeella kyseinen ruutu sijaitsee.
103      * @return naapurissa olevien miinojen määrä.
104      */
105     public int montaMiinaaNaapurissa(int rivi, int sarake) {
106         int arvo = 0;
107         for (int i = -1; i <= 1; i++) {
108             for (int k = -1; k <= 1; k++) {
109                 int uusiRivi = rivi - i;
110                 int uusiSarake = sarake - k;
111                 if (uusiRivi == rivi && uusiSarake == sarake) {
112                     continue;
113                 } else if (uusiRivi >= 0 && uusiRivi < koko && uusiSarake >= 0 && uusiSarake < koko) {
114                     if (ruudut[uusiRivi][uusiSarake].isOnkoMiina()) {
115                         arvo++;
116                     }
117                 }
118             }
119         }
120         return arvo;
121     }
122 
123     public int getKoko() {
124         return koko;
125     }
126 
127     public Ruutu[][] getRuudut() {
128         return ruudut;
129     }
130 
131 }
