1   package fi.jonna.mehtonen.domain;

3   import fi.jonna.mehtonen.domain.Ruutu;
4   import java.util.Random;
5   
6   /**
7    * Pelilauta luokka alustaa itse pelilaudan luomalla ruudut, asettamalla niihin
8    * arvot ja asettamalla miinat.
9    */
10  public class Pelilauta {
11  
12      private int koko;
13      private Ruutu[][] ruudut;
14      private Random rnd;
15  
16      /**
17       * Konstruktorissa asetetaan laudan koko halutuksi, luodaan random ja Ruutu
18       * -matriisi oliot sekä alustetaan pelilauta.
19       *
20       * @param laudanKoko Pelilaudan haluttu koko.
21       * @param ruudut RuudukonAlustuksessa luotu ruudukko, joka sisältää miinat.
22       */
23      public Pelilauta(int laudanKoko, Ruutu[][] ruudut) {
24          this.koko = laudanKoko;
25          this.ruudut = ruudut;
26          this.rnd = new Random();
27          asetaRuuduilleArvot();
28      }
29  
30      /**
31       * Jos ensimmäinen ruutu, joka avataan sisältää miinan, niin miinalle
32       * arvotaan uusi paikka ja peli jatkuu. Kayttaja ei huomaa siirtoa.
33       *
34       * @param rivi kertoo, millä rivillä korvattava miina sijaitsee.
35       * @param sarake kertoo, millä sarakkeella korvattava miina sijaitsee.
36       */
37      public void siirraMiina(int rivi, int sarake) {
38          boolean siirretty = false;
39          while (!siirretty) {
40              int uusiRivi = rnd.nextInt(koko);
41              int uusiSarake = rnd.nextInt(koko);
42              if (!ruudut[uusiRivi][uusiSarake].isOnkoMiina()) {
43                  ruudut[rivi][sarake].poistaMiina();
44                  ruudut[uusiRivi][uusiSarake].setMiina();
45                  asetaRuuduilleArvot();
46                  siirretty = true;
47              }
48          }
49      }
50  
51      /**
52       * Asetetaan ruuduille arvoiksi se luku, joka vastaa naapuriruuduissa
53       * olevien miinojen maara.
54       */
55      public void asetaRuuduilleArvot() {
56          for (int k = 0; k < koko; k++) {
57              for (int j = 0; j < koko; j++) {
58                  if (!ruudut[k][j].isOnkoMiina()) {
59                      ruudut[k][j].setArvo(montaMiinaaNaapurissa(k, j));
60                  }
61              }
62          }
63  
64      }
65  
66      /**
67       * Lasketaan, kuinka monta miinaa on ruudun naapureissa.
68       *
69       * @param rivi kertoo, millä rivillä kyseinen ruutu sijaitsee.
70       * @param sarake kertoo, millä sarakkeella kyseinen ruutu sijaitsee.
71       * @return naapurissa olevien miinojen määrä.
72       */
73      public int montaMiinaaNaapurissa(int rivi, int sarake) {
74          int arvo = 0;
75          for (int i = -1; i <= 1; i++) {
76              for (int k = -1; k <= 1; k++) {
77                  int uusiRivi = rivi - i;
78                  int uusiSarake = sarake - k;
79                  if (uusiRivi == rivi && uusiSarake == sarake) {
80                      continue;
81                  } else if (uusiRivi >= 0 && uusiRivi < koko && uusiSarake >= 0 && uusiSarake < koko) {
82                      if (ruudut[uusiRivi][uusiSarake].isOnkoMiina()) {
83                          arvo++;
84                      }
85                  }
86              }
87          }
88          return arvo;
89      }
90  
91      public int getKoko() {
92          return koko;
93      }
94  
95      public Ruutu[][] getRuudut() {
96          return ruudut;
97      }
98  
99  }
