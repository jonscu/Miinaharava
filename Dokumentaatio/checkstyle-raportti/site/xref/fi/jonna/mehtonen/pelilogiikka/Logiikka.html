9   /**
10   * Logiikka luokassa tarkastellaan, mitä tehdään, kun painetaan hiiren oikeata
11   * tai vasenta klikkiä sekä onko peli voitettu tai hävitty. Lisäksi Logiikka
12   * luokka tarkistaa, avataanko naapuri ruudut.
13   */
14  public class Logiikka {
15  
16      private int koko, miinoja;
17      private Ruutu[][] ruudut;
18      private JButton[][] nappulat;
19      private Pelilauta lauta;
20      private int siirrot = 0;
21      private int lippujaKaytetty = 0;
22      private MiinaPaneeli paneeli;
23  
24      /**
25       * Logiikka luokan konstruktorissa saadaan selville luodun Pelilaudan koko,
26       * itse Pelilauta ja sen Ruudut, miinojen määrä, MiinaPaneelissa luotu
27       * JButton matriisi ja itse MiinaPaneeli.
28       *
29       * @param koko Luodun pelilaudan koko.
30       * @param lauta Luotu pelilauta.
31       * @param miinoja Miinojen määrä pelilaudassa.
32       * @param nappulat MiinaPaneelissa luodut JButton matriisi.
33       * @param paneeli Itse MiinaPaneeli.
34       */
35      public Logiikka(int koko, Pelilauta lauta, int miinoja, JButton[][] nappulat, MiinaPaneeli paneeli) {
36          this.lauta = lauta;
37          this.ruudut = lauta.getRuudut();
38          this.koko = koko;
39          this.miinoja = miinoja;
40          this.nappulat = nappulat;
41          this.paneeli = paneeli;
42      }
43  
44      /**
45       * Tarkistetaan, onko peli voitettu. Peli on voitettu, jos kaikki
46       * miinattomat ruudut ovat avattu.
47       *
48       * @return true, jos peli on voitettu, muulloin false.
49       */
50      public Boolean voitto() {
51          int ruutujaAvattu = 0;
52  
53          for (int k = 0; k < koko; k++) {
54              for (int j = 0; j < koko; j++) {
55                  if (ruudut[k][j].isOnkoAvattu() && !ruudut[k][j].isOnkoMiina()) {
56                      ruutujaAvattu++;
57                  }
58              }
59          }
60          if (ruutujaAvattu == (koko * koko) - miinoja) {
61              return true;
62          } else {
63              return false;
64          }
65      }
66  
67      /**
68       * Tarkistetaan, onko peli hävitty. Peli hävitään, jos käyttäjä avaa ruudun,
69       * jossa on miina.
70       *
71       * @return true, jos peli hävittiin, muulloin false.
72       */
73      public Boolean havio() {
74          for (int k = 0; k < koko; k++) {
75              for (int j = 0; j < koko; j++) {
76                  if (ruudut[k][j].isOnkoMiina() && ruudut[k][j].isOnkoAvattu()) {
77                      return true;
78                  }
79              }
80          }
81          return false;
82      }
83  
84      /**
85       * Tarkistetaan painettiinko oikeeta vai vasenta hiiren nappulaa. Jos
86       * klikattiin vasenta nappulaa, merkataan ruutu avatuksi ja tarkistetaan,
87       * voitettiinko tai havittiinko peli. Jos klikattiin oikeata nappulaa, niin
88       * laitetaan ruutuun lippu, mikali siina ei ennestaan ollut lippua. Mikäli
89       * oli, niin lippu otetaan pois, jolloin nappi on avaamaton.
90       *
91       * @param tapahtuma Hiiren klikkauksesta aiheutuva MouseEvent.
92       */
93      public void avataanRuutu(MouseEvent tapahtuma) {
94          Component nappula = tapahtuma.getComponent();
95          if (nappula.isEnabled() && tapahtuma.getButton() == 1) {
96              for (int k = 0; k < koko; k++) {
97                  for (int j = 0; j < koko; j++) {
98                      if (nappula == nappulat[k][j]) {
99                          avataanRuutuTaiSiirretaanMiina(k, j);
100                         break;
101                     }
102                 }
103             }
104         } else if (tapahtuma.getButton() == 3) {
105             for (int k = 0; k < koko; k++) {
106                 for (int j = 0; j < koko; j++) {
107                     if (nappula == nappulat[k][j]) {
108                         laitetaanTaiPoistetaanLippu(k, j);
109                         break;
110                     }
111                 }
112             }
113 
114         }
115     }
116 
117     /**
118      * Jos ensimmäisenä pelaaja avaa miinan, niin miina siirretään. Muussa
119      * tapauksessa Ruutu avataan ja paljastetaan naapurissa olevien miinojen
120      * lukumäärä, mikäli ruutua ei ole ennen avattu.
121      *
122      * @param k Ruudun rivi sijainti.
123      * @param j Ruudun sarake sijainti.
124      */
125     public void avataanRuutuTaiSiirretaanMiina(int k, int j) {
126         if (ruudut[k][j].isOnkoMiina() && siirrot == 0 && !ruudut[k][j].isOnkoLippu()) {
127             lauta.siirraMiina(k, j);
128             paneeli.merkataanRuutuAvatuksi(nappulat[k][j], k, j);
129             avaaNaapurit(nappulat[k][j], k, j, nappulat);
130             siirrot++;
131         } else if (!ruudut[k][j].isOnkoLippu()) {
132             paneeli.merkataanRuutuAvatuksi(nappulat[k][j], k, j);
133             if (ruudut[k][j].getArvo() == 0) {
134                 avaaNaapurit(nappulat[k][j], k, j, nappulat);
135             }
136             siirrot++;
137             tarkistaPelinLoppuminen();
138         }
139     }
140 
141     /**
142      * Jos ruudussa ei ennestään ole lippua, niin siihen laitetaan lippu. Jos
143      * ruudussa oli ennestään lippu, niin se otetaan pois.
144      *
145      * @param k Ruudun rivi sijainti.
146      * @param j Ruudun sarake sijainti.
147      */
148     public void laitetaanTaiPoistetaanLippu(int k, int j) {
149         if (!ruudut[k][j].isOnkoLippu() && lippujaKaytetty != miinoja && !ruudut[k][j].isOnkoAvattu()) {
150             lippujaKaytetty++;
151             paneeli.asetaLippu(nappulat[k][j], k, j);
152         } else if (ruudut[k][j].isOnkoLippu()) {
153             lippujaKaytetty--;
154             paneeli.poistaLippu(nappulat[k][j], k, j);
155         }
156     }
157 
158     /**
159      * Metodi kutsuu voitaPeli, jos peli voitettiin. Jos peli hävittiin, niin se
160      * kutsuu haviaPeli metodia.
161      */
162     public void tarkistaPelinLoppuminen() {
163         if (voitto()) {
164             paneeli.voitaPeli();
165         } else if (havio()) {
166             paneeli.haviaPeli();
167         }
168     }

170     /**
171      * Mikali klikatun ruudun yhdessakaan naapuriruudussa ei ole miinaa, peli
172      * paljastaa ensin kaikki siihen yhteydessa olevat ruudut, joiden naapurissa
173      * ei myoskaan ole miinaa, sekä lisaksi taman paljastetun alueen reunalla
174      * olevat ruudut, joiden naapurissa on miina.
175      *
176      * @param nappula JButton, jota on klikattu.
177      * @param rivi nappulan rivi sijainti.
178      * @param sarake nappulan sarake sijainti.
179      * @param nappulat laudan kaikki nappulat matriisissa.
180      */
181     public void avaaNaapurit(JButton nappula, int rivi, int sarake, JButton[][] nappulat) {
182         for (int i = -1; i <= 1; i++) {
183             for (int k = -1; k <= 1; k++) {
184                 int uusiRivi = rivi - i;
185                 int uusiSarake = sarake - k;
186                 if (uusiRivi == rivi && uusiSarake == sarake) {
187                     continue;
188                 } else if (uusiRivi >= 0 && uusiRivi < koko && uusiSarake >= 0 && uusiSarake < koko) {
189                     if (!ruudut[uusiRivi][uusiSarake].isOnkoAvattu() && ruudut[uusiRivi][uusiSarake].getArvo() == 0 && !ruudut[uusiRivi][uusiSarake].isOnkoMiina()) {
190                         paneeli.merkataanRuutuAvatuksi(nappulat[uusiRivi][uusiSarake], uusiRivi, uusiSarake);
191                         avaaNaapurit(nappulat[uusiRivi][uusiSarake], uusiRivi, uusiSarake, nappulat);
192                     } else if (ruudut[rivi][sarake].getArvo() == 0 && !ruudut[uusiRivi][uusiSarake].isOnkoAvattu() && ruudut[uusiRivi][uusiSarake].getArvo() != 0 && !ruudut[uusiRivi][uusiSarake].isOnkoMiina()) {
193                         paneeli.merkataanRuutuAvatuksi(nappulat[uusiRivi][uusiSarake], uusiRivi, uusiSarake);
194                     }
195                 }
196             }
197         }
198     }
199 
200     public int getLippujaKaytetty() {
201         return lippujaKaytetty;
202     }
